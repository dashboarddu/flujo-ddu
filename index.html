const CORS_HEADERS = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, PATCH, DELETE, OPTIONS", "Access-Control-Allow-Headers": "Content-Type, Authorization", }; function jsonResponse(data, status = 200) { return new Response(JSON.stringify(data), { status, headers: { ...CORS_HEADERS, "Content-Type": "application/json" }, }); } async function signToken(email, secret) { const encoder = new TextEncoder(); const key = await crypto.subtle.importKey( "raw", encoder.encode(secret), { name: "HMAC", hash: "SHA-256" }, false, ["sign"] ); const payload = `${email}:${Date.now()}`; const sig = await crypto.subtle.sign("HMAC", key, encoder.encode(payload)); const b64 = btoa(String.fromCharCode(...new Uint8Array(sig))); return `${btoa(payload)}.${b64}`; } async function verifyToken(token, secret) { try { const [payloadB64, sigB64] = token.split("."); const payload = atob(payloadB64); const [email, ts] = payload.split(":"); // Token válido por 8 horas if (Date.now() - parseInt(ts) > 8 * 60 * 60 * 1000) return null; const encoder = new TextEncoder(); const key = await crypto.subtle.importKey( "raw", encoder.encode(secret), { name: "HMAC", hash: "SHA-256" }, false, ["verify"] ); const sig = Uint8Array.from(atob(sigB64), (c) => c.charCodeAt(0)); const valid = await crypto.subtle.verify("HMAC", key, sig, encoder.encode(payload)); return valid ? email : null; } catch { return null; } } export default { async fetch(request, env) { const url = new URL(request.url); const path = url.pathname; if (request.method === "OPTIONS") { return new Response(null, { headers: CORS_HEADERS }); } // ── RUTA: /auth/login ──────────────────────────────────── // Solo verifica que el correo esté en la lista y entrega token if (path === "/auth/login" && request.method === "POST") { const { email } = await request.json(); const normalizedEmail = (email || "").toLowerCase().trim(); if (!normalizedEmail || !ALLOWED_EMAILS.includes(normalizedEmail)) { return jsonResponse({ error: "Correo no autorizado." }, 403); } const token = await signToken(normalizedEmail, env.EMAIL_SECRET); return jsonResponse({ ok: true, token, email: normalizedEmail }); } // ── VERIFICACIÓN DE TOKEN ──────────────────────────────── const authHeader = request.headers.get("Authorization") || ""; const token = authHeader.replace("Bearer ", "").trim(); const userEmail = await verifyToken(token, env.EMAIL_SECRET); if (!userEmail) { return jsonResponse({ error: "No autorizado." }, 401); } // ── RUTA: /api/records ─────────────────────────────────── if (path === "/api/records") { const airtableBase = `https://api.airtable.com/v0/${env.AIRTABLE_BASE_ID}/Documentos`; const headers = { Authorization: `Bearer ${env.AIRTABLE_API_KEY}` }; if (request.method === "GET") { let records = []; let offset = null; do { const params = new URLSearchParams({ pageSize: "100" }); if (offset) params.set("offset", offset); const res = await fetch(`${airtableBase}?${params}`, { headers }); const data = await res.json(); if (data.records) records = records.concat(data.records); offset = data.offset || null; } while (offset); return jsonResponse({ records }); } if (request.method === "POST") { const body = await request.json(); const res = await fetch(airtableBase, { method: "POST", headers: { ...headers, "Content-Type": "application/json" }, body: JSON.stringify(body), }); const data = await res.json(); return jsonResponse(data, res.status); } } // ── RUTA: /api/records/:id ─────────────────────────────── const recordMatch = path.match(/^\/api\/records\/(.+)$/); if (recordMatch) { const recordId = recordMatch[1]; const airtableUrl = `https://api.airtable.com/v0/${env.AIRTABLE_BASE_ID}/Documentos/${recordId}`; const headers = { Authorization: `Bearer ${env.AIRTABLE_API_KEY}`, "Content-Type": "application/json", }; if (request.method === "PATCH") { const body = await request.json(); const res = await fetch(airtableUrl, { method: "PATCH", headers, body: JSON.stringify(body) }); const data = await res.json(); return jsonResponse(data, res.status); } if (request.method === "DELETE") { const res = await fetch(airtableUrl, { method: "DELETE", headers }); const data = await res.json(); return jsonResponse(data, res.status); } } return jsonResponse({ error: "Ruta no encontrada." }, 404); }, };
